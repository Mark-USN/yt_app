lib.display_field

A DisplayField represents a single piece of information to be displayed in the UI. It combines data's title, a seperator, locale formatted metadata and possibly units into a single string that can be easily displayed in the UI. It holds a reference to a StringVar that the UI widget binds to, and knows how to format its value for display based on its metadata. It also holds and can return the raw value for use in calculations or other logic.

 33class DisplayField:
 34    """ A DisplayField represents a single piece of information to be displayed in the UI.
 35        It lncludes the label, separator, units, and formatting options for the value.
 36        It holds a reference to a StringVar that the UI widget binds to, and knows how to 
 37        format its value for display based on its metadata.
 38    """
 39    ctx: RunContextStore
 40    label: str
 41    var: StringVar | None = None
 42    sep: str = ": "
 43    units: str = ""
 44    decimals: int = 2
 45    is_int: bool = False
 46    _value: float | int | str | None = field(default=None, init=False, repr=False)
 47
 48    def set(self, value: float | int | str | None) -> None:
 49        """ Set the underlying value of the field and update the StringVar if it exists. 
 50            Args:
 51                value: The new value to set for the field. Can be a float, int, or str.
 52        """
 53        self._value = value
 54        if self.var is not None:
 55            self.var.set(self.render())
 56
 57    def get(self) -> float | int | str | None:
 58        """ Get the underlying value of the field. 
 59            Returns:
 60                The raw value of the field, which can be a float, int, str, or None.
 61        """                
 62        return self._value
 63
 64    def render(self) -> str:
 65        """ Render the field's value as a string for display. 
 66            Returns:
 67                A string representing the formatted value of the field.
 68        """
 69        if self._value is None:
 70            value_text = ""
 71        elif isinstance(self._value, float):
 72            if self.is_int:
 73                value_text = self.ctx.format_number(
 74                    value=self._value,
 75                    decimals=0,
 76                    as_int=True,
 77                )
 78            else:
 79                value_text = self.ctx.format_number(
 80                    value=self._value,
 81                    decimals=self.decimals,
 82                )
 83        elif isinstance(self._value, int):
 84            value_text = self.ctx.format_number(
 85                value=self._value,
 86                decimals=0,
 87                as_int=True,
 88            )
 89        else:
 90            value_text = str(self._value)
 91
 92        if self.units and value_text:
 93            return f"{self.label}{self.sep}{value_text} {self.units}"
 94        return f"{self.label}{self.sep}{value_text}"

104        return cls(

106            label=data["label"],
107            var=data["var"],
108            sep=data.get("sep", ": "),
109            units=data.get("units", ""),
110            decimals=data.get("decimals", 2),
111            is_int=data.get("is_int", False),
112        )

114def format_hms(seconds: float | int | None) -> str:
115    """ Format a duration in seconds into a human-readable string (H:MM:SS or M:SS). 
116        Args:
117            seconds: The duration in seconds to format. Can be a float, int, or None.
118        Returns:
119                A string representing the formatted duration.
120        Notes:
121            Solves a couple of issues with the built-in time formatting functions:
122                Avoids leading zeros for hours and minutes, but always uses two digits for seconds.
123                Does not wrap at 24 hours, so durations longer than 24 hours will show the total
124                hours.
125        """
126    if seconds is None:
127        return ""
128
129    total = int(round(seconds))
130    sign = "-" if total < 0 else ""
131    total = abs(total)
132
133    hours, rem = divmod(total, 3600)
134    minutes, secs = divmod(rem, 60)
135
136    if hours:
137        return f"{sign}{hours}:{minutes:02}:{secs:02}"
138    return f"{sign}{minutes}:{secs:02}"

164class FileSizeField(DisplayField):
165    """ A specialized DisplayField for formatting file sizes in bytes into human-readable strings
166        with appropriate units. 
167    """
168
169    def get(self) -> int | None:
170        """ Get the underlying value of the field as an integer representing bytes. 
171            Returns:
172                The file size in bytes as an integer, or None if the value is not set.
173        """
174        value = self._value
175        return int(value) if value is not None else None
176
177    def render(self) -> str:
178        """ Render the field's value as a formatted file size string with appropriate units.
179            Returns:
180                A string representing the formatted file size, including units (bytes, KB, or MB).
181        """
182        size = self.get()
183        kb = 1024
184        mb = kb * 1024
185        if size is None:
186            return "0"
187        if size < kb:
188            return f"{self.label}{self.sep}{self.ctx.format_number(size, as_int=True)} bytes"
189        if size < mb:
190            return f"{self.label}{self.sep}{self.ctx.format_number(size / kb, decimals=3)} KB"
191        return f"{self.label}{self.sep}{self.ctx.format_number(size / mb, decimals=3)} MB"

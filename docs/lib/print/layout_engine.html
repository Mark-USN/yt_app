lib.print.layout_engine

Shared line-layout logic for plain-text reports.

This module contains the output-target-independent formatting rules used by both the printer and PDF backends.

Supported source item types:

  • LineItem: A single source line that may wrap by words.
  • ParagraphItem: A list of source lines, each wrapped independently.
  • BlocksItem: A list of intact block strings packed together onto lines without splitting any individual block.

 33def is_long_word(word: str, max_width: float, measurer: TextMeasurer) -> bool:
 34    """ Determine if a single word exceeds the available width.
 35        Args:
 36            word: The word to check.
 37            max_width: The maximum allowable width.
 38            measurer: The text measurer to use. 
 39        Returns:
 40            True if the word exceeds the max width, False otherwise.
 41    
 42    """
 43    return measurer.measure(word) > max_width

 54def split_long_word(
 55    text: str,
 56    max_width: float,
 57    measurer: TextMeasurer,
 58) -> list[str]:
 59    """ Split a long string into width-constrained chunks using binary search.
 60        Args:
 61            text: The long string to split.
 62            max_width: The maximum allowable width.
 63            measurer: The text measurer to use.
 64        Returns:
 65            A list of width-constrained chunks.
 66    """
 67
 68    result: list[str] = []
 69    i = 0
 70    n = len(text)
 71
 72    while i < n:
 73        lo = i + 1
 74        hi = n
 75        best = lo
 76
 77        # Binary search for the longest substring that fits
 78        while lo <= hi:
 79            mid = (lo + hi) // 2
 80            chunk = text[i:mid]
 81            width = measurer.measure(chunk)
 82
 83            if width <= max_width:
 84                best = mid
 85                lo = mid + 1
 86            else:
 87                hi = mid - 1
 88
 89        # Prevent infinite loop (very narrow width case)
 90        if best == i:
 91            best = i + 1
 92
 93        result.append(text[i:best])
 94        i = best
 95
 96    return result

101def wrap_text_words(
102    text: str,
103    *,
104    measurer: TextMeasurer,
105    max_width: float,
106    continuation_indent: float,
107) -> list[RenderLine]:
108    """ Wrap a logical source line into one or more rendered lines.
109        Args:
110            text: The logical source line to wrap.
111            measurer: Width measurer for the active backend.
112            max_width: Maximum usable width from the left margin to the right margin.
113            continuation_indent: Horizontal offset applied to continuation lines.
114        Returns:
115            A list of wrapped rendered lines.
116
117        Wrapping is performed at word boundaries. The first rendered line starts
118        at the normal left margin. Continuation lines are indented by
119        ``continuation_indent``.
120        Notes
121        -----
122        If a single word is wider than the available width, it is emitted as-is
123        on its own line rather than being split mid-word.
124    """
125    stripped = text.strip()
126    if not stripped:
127        return [RenderLine(0.0, "")]
128
129    words = stripped.split()
130    lines: list[RenderLine] = []
131    current_words: list[str] = []
132    current_indent = 0.0
133
134    for word in words:
135
136        word_slices: list[str] = []
137        candidate_words = [*current_words, word]
138        candidate_text = " ".join(candidate_words)
139        candidate_width = current_indent + measurer.measure(candidate_text)
140
141        if is_long_word(word, max_width, measurer):
142            word_slices:list[str] = split_long_word(candidate_text, max_width, measurer)
143            lines.append(RenderLine(max_width, chunk) for chunk in word_slices[:-2])
144            current_words = []
145            word = word_slices[-1]
146            candidate_words = [word]
147            candidate_text = " ".join(candidate_words)
148            candidate_width = current_indent + measurer.measure(candidate_text)
149
150        if current_words and candidate_width > max_width:
151            lines.append(RenderLine(current_indent, " ".join(current_words)))
152            current_words = [word]
153            current_indent = continuation_indent
154            continue
155
156        current_words = candidate_words
157
158    if current_words:
159        lines.append(RenderLine(current_indent, " ".join(current_words)))
160
161    return lines

164def wrap_centered_text(
165    text: str,
166    *,
167    measurer: TextMeasurer,
168    max_width: float,
169) -> list[RenderLine]:
170    """ Wrap a logical source line into one or more rendered lines.
171        Args:
172            text: The logical source line to wrap and center.
173            measurer: Width measurer for the active backend.
174            max_width: Maximum usable width from the left margin to the right margin.
175        Returns:
176            A list of wrapped rendered lines.
177
178        Wrapping is performed at word boundaries. The first rendered line starts
179        at the normal left margin. Continuation lines are indented by
180        ``continuation_indent``.
181        Notes
182        -----
183        If a single word is wider than the available width, it is emitted as-is
184        on its own line rather than being split mid-word.
185    """
186    stripped = text.strip()
187    if not stripped:
188        return [RenderLine(0.0, "")]
189
190    words = stripped.split()
191    lines: list[RenderLine] = []
192    current_words: list[str] = []
193    total_width = measurer.measure(text)
194    total_lines = math.ceil(total_width / max_width)
195    avg_line_width = total_width / total_lines
196    max_line_width = min(max_width, avg_line_width * 1.25)  # Allow some variance in line widths
197
198    for word in words:
199        word_slices: list[str] = []
200        candidate_words = [*current_words, word]
201        candidate_text = " ".join(candidate_words)
202        if is_long_word(word, max_line_width, measurer):
203            word_slices = split_long_word(candidate_text, max_line_width, measurer)
204            lines.extend(RenderLine(
205                (max_width - measurer.measure(slice)) / 2, slice) for slice in word_slices[:-2])
206            current_words = []
207            word = word_slices[-1]
208            candidate_words = [word]
209            candidate_text = " ".join(candidate_words)
210
211        # candidate_width = current_indent + measurer.measure(candidate_text)
212        candidate_width = measurer.measure(candidate_text)
213
214        if current_words and candidate_width > max_line_width:
215            indent = (max_width - measurer.measure(" ".join(current_words))) / 2
216            lines.append(RenderLine(indent, " ".join(current_words)))
217            current_words = [word]
218            # current_indent = continuation_indent
219            continue
220
221        current_words = candidate_words
222
223    if current_words:
224        current_text = " ".join(current_words)
225        indent = (max_width - measurer.measure(current_text)) / 2
226        lines.append(RenderLine(indent, current_text))
227
228    return lines

231def blocks_to_lines(
232    blocks: list[str],
233    *,
234    measurer: TextMeasurer,
235    max_width: float,
236    separator: str,
237) -> list[RenderLine]:
238    """ Pack intact block strings onto rendered lines.
239        Args:
240            blocks: The intact block strings to pack.
241            measurer: Width measurer for the active backend.
242            max_width: Maximum usable width from the left margin to the right margin.
243            separator: String inserted between adjacent blocks on the same output line.
244        Returns:
245            A list of packed rendered lines.
246
247        Each block remains whole. Blocks are combined with ``separator`` until
248        adding the next block would exceed ``max_width``. A new rendered line is
249        then started.
250
251        Notes
252        -----
253        If a single block exceeds the available width by itself, it is emitted
254        on a line of its own rather than being split.
255    """
256    if not blocks:
257        return []
258
259    lines: list[RenderLine] = []
260    current = blocks[0]
261
262    for block in blocks[1:]:
263        candidate = f"{current}{separator}{block}"
264        if measurer.measure(candidate) <= max_width:
265            current = candidate
266        else:
267            lines.append(RenderLine(0.0, current))
268            current = block
269
270    lines.append(RenderLine(0.0, current))
271    return lines

274def expand_items_to_lines(
275    items: Iterable[RenderItem],
276    *,
277    measurer: TextMeasurer,
278    layout: PageLayout,
279) -> list[RenderLine]:
280    """ Convert mixed input items into fully laid-out rendered lines.
281        Args:
282            items: The mixed content items to format.
283            measurer: Width measurer for the active backend.
284            layout: Page layout containing usable width, indent, and separator settings.
285        Returns:
286            A list of fully laid-out rendered lines.
287    """
288    output: list[RenderLine] = []
289    max_width = layout.usable_width
290
291    for item in items:
292        if isinstance(item, LineItem):
293            output.extend(
294                wrap_text_words(
295                    item.text,
296                    measurer=measurer,
297                    max_width=max_width,
298                    continuation_indent=layout.wrap_indent,
299                )
300            )
301        elif isinstance(item, CenteredLineItem):
302            output.extend(
303                wrap_centered_text(
304                    item.text,
305                    measurer=measurer,
306                    max_width=max_width,
307                )
308            )
309        elif isinstance(item, ParagraphItem):
310            for line in item.lines:
311                output.extend(
312                    wrap_text_words(
313                        line,
314                        measurer=measurer,
315                        max_width=max_width,
316                        continuation_indent=layout.wrap_indent,
317                    )
318                )
319        elif isinstance(item, BlocksItem):
320            output.extend(
321                blocks_to_lines(
322                    item.blocks,
323                    measurer=measurer,
324                    max_width=max_width,
325                    separator=layout.block_separator,
326                )
327            )
328        else:
329            msg = f"Unsupported render item type: {type(item).__name__}"
330            raise TypeError(msg)
331
332    return output
